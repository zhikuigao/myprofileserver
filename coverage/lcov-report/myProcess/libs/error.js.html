var generateError = function(name, status, message){
    var c = function(n, s, m){
        var temp = Error.call(this);
        temp.name = this.name = name || n;
        this.status = status || s;
        this.message = message || m;
        this.stack = temp.stack;
    };
 
    c.prototype = Object.create(Error.prototype, {
        constructor: {
            value: c,
            writable: true,
            configurable: true
        }
    });
 
    return c;
};
