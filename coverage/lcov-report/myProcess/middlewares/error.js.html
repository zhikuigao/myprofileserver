let _ = require('lodash');
 
module.exports = error;
 
 
function error(opts){
    const env = process.env.NODE_ENV || 'development';
 
    return function *error(next){
        try{
            yield next;
        }catch(err){
            this.status = err.status || 500;
            this.res.statusMessage = err.name || 'server error';
            console.log(err.stack);
 
            switch (this.accepts('html', 'text', 'json')) {
                case 'json':
                    this.type = 'application/json';
                    if(err.name == 'ParameterError' || err.name == 'ValidationError'){
                        this.status =531;
                        this.res.statusMessage = 'ParameterError';
                        let body = (err.name == 'ParameterError')  && err.message;
 
                        if(err.name == 'ValidationError'){
                            body = _.mapValues(err.errors,(value)=>{ // err.errors : { color: {message: '', kind: '', path: '', value: ''}...... }
                                return  value.message;   // ret : [{skip: 'not number'}, {username: 'required'}]
                            })
                        }
                        this.body = body;
                    }else{
                        this.body = {message: err.message};
                    }
                    break;
                case 'html':
                    this.type = 'text/html';
                    this.body = err.message;
                    break;
                default :
                    this.type = 'text/plain';
                    this.body = err.message;
                    break;
            }
        }
    };
};
